        // --- Theme Engine ---
        function initTheme() {
            if (localStorage.getItem('ninjaTheme') === 'dark' || (!('ninjaTheme' in localStorage) && window.matchMedia('(prefers-color-scheme: dark)').matches)) {
                isDarkMode = true;
                document.documentElement.classList.add('dark');
            } else {
                isDarkMode = false;
                document.documentElement.classList.remove('dark');
            }
            updateChartColors();
        }

        function toggleTheme() {
            isDarkMode = !isDarkMode;
            if (isDarkMode) {
                document.documentElement.classList.add('dark');
                localStorage.setItem('ninjaTheme', 'dark');
                showToast("Genjutsu Activated: Itachi Theme", "success", "fa-eye");
            } else {
                document.documentElement.classList.remove('dark');
                localStorage.setItem('ninjaTheme', 'light');
                showToast("Flying Raijin: Minato Theme", "success", "fa-bolt");
            }
            updateChartColors();
            generateAIInsight(); // Update persona text
        }

        // --- UI Utilities (Toast & Modal) ---
        function showToast(message, type = 'success', icon = 'fa-check') {
            const container = document.getElementById('toast-container');
            const toast = document.createElement('div');
            
            const bgClass = isDarkMode ? 'bg-slate-800 border-slate-700' : 'bg-white border-slate-200';
            const iconColor = type === 'success' ? 'text-emerald-500' : 'text-red-500';
            
            toast.className = `toast-enter flex items-center gap-3 px-4 py-3 rounded-xl shadow-lg border ${bgClass} pointer-events-auto max-w-sm`;
            toast.innerHTML = `
                <i class="fa-solid ${icon} ${iconColor} text-lg"></i>
                <p class="text-sm font-medium text-slate-800 dark:text-slate-200">${message}</p>
            `;
            
            container.appendChild(toast);
            setTimeout(() => {
                toast.style.opacity = '0';
                toast.style.transform = 'translateY(20px)';
                toast.style.transition = 'all 0.3s ease';
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }

        function showConfirm(title, message, onConfirm) {
            const modal = document.getElementById('custom-modal');
            const content = document.getElementById('modal-content');
            
            document.getElementById('modal-title').textContent = title;
            document.getElementById('modal-body').textContent = message;
            
            modal.classList.remove('hidden');
            // Trigger reflow
            void modal.offsetWidth;
            modal.classList.remove('opacity-0');
            content.classList.remove('scale-95');
            
            const cleanup = () => {
                modal.classList.add('opacity-0');
                content.classList.add('scale-95');
                setTimeout(() => modal.classList.add('hidden'), 300);
                document.getElementById('modal-confirm').onclick = null;
                document.getElementById('modal-cancel').onclick = null;
            };

            document.getElementById('modal-cancel').onclick = cleanup;
            document.getElementById('modal-confirm').onclick = () => {
                onConfirm();
                cleanup();
            };
        }

        // --- Helper Functions ---
        function formatDuration(minutes) {
            const h = Math.floor(minutes / 60);
            const m = minutes % 60;
            if (h > 0 && m > 0) return `${h}h ${m}m`;
            if (h > 0) return `${h}h`;
            return `${m}m`;
        }
        
        function getTodayDateString() {
            const d = new Date();
            return new Date(d.getTime() - (d.getTimezoneOffset() * 60000)).toISOString().split('T')[0];
        }

        function formatDisplayDate(dateString) {
            if (dateString === getTodayDateString()) return "Today";
            return new Date(dateString).toLocaleDateString('en-US', { weekday: 'short', month: 'short', day: 'numeric' });
        }

        function formatTime12Hour(timeStr) {
            if(!timeStr) return '';
            const [h, m] = timeStr.split(':');
            const date = new Date();
            date.setHours(h, m, 0);
            return date.toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit', hour12: true });
        }

            localStorage.setItem('ninjaTrackerData', JSON.stringify(activities));
        }

        function clearAllData() {
            showConfirm("Wipe Database", "Are you sure you want to permanently delete all mission logs? This cannot be undone.", () => {
                activities = [];
                saveData();
                updateDashboard();
                showToast("All logs erased.", "success", "fa-eraser");
            });
        }

        function exportToExcel() {
            if (activities.length === 0) {
                showToast("No data to export.", "error", "fa-triangle-exclamation");
                return;
            }
            
            // Format data for excel
            const exportData = activities.map(a => ({
                Date: a.date,
                Name: a.name,
                Category: a.category,
                StartTime: formatTime12Hour(a.startTime),
                EndTime: formatTime12Hour(a.endTime),
                DurationMinutes: a.duration,
                Description: a.description
            }));

            const ws = XLSX.utils.json_to_sheet(exportData);
            const wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, "Mission Logs");
            XLSX.writeFile(wb, "Ninja_Tracker_Export.xlsx");
            
            showToast("Excel exported successfully!", "success", "fa-download");
        }

        function handleAddActivity(e) {
            e.preventDefault();
            
            const nameInput = document.getElementById('activity-name').value.trim();
            const dateInput = document.getElementById('activity-date').value;
            const catInput = document.getElementById('activity-category').value;
            const startInput = document.getElementById('activity-start').value;
            const endInput = document.getElementById('activity-end').value;
            const descInput = document.getElementById('activity-desc').value.trim();

            const startDateTime = new Date(`${dateInput}T${startInput}`);
            let endDateTime = new Date(`${dateInput}T${endInput}`);
            
            if (endDateTime < startDateTime) endDateTime.setDate(endDateTime.getDate() + 1);

            const durationMinutes = Math.round((endDateTime - startDateTime) / 60000);

            if (durationMinutes <= 0) {
                showToast("Time paradox! End time must be after start.", "error", "fa-clock");
                return;
            }

            const newActivity = {
                id: Date.now().toString(),
                name: nameInput,
                category: catInput,
                date: dateInput,
                startTime: startInput,
                endTime: endInput,
                duration: durationMinutes,
                description: descInput,
                timestamp: new Date().toISOString()
            };

            activities.push(newActivity);
            saveData();
            
            if (selectedDate !== dateInput) {
                selectedDate = dateInput;
                document.getElementById('global-date-filter').value = selectedDate;
            }
            
            document.getElementById('activity-start').value = '';
            document.getElementById('activity-end').value = '';
            document.getElementById('activity-desc').value = '';
            document.getElementById('activity-name').focus();

            updateDashboard();
            showToast("Mission recorded successfully.", "success", "fa-scroll");
        }

        function deleteActivity(id) {
            showConfirm("Delete Log", "Remove this mission from the archives?", () => {
                activities = activities.filter(a => a.id !== id);
                saveData();
                updateDashboard();
                showToast("Mission deleted.", "success", "fa-trash");
            });
        }

        function updateStats() {
            const dayActivities = activities.filter(a => a.date === selectedDate);
            const totalMins = dayActivities.reduce((sum, a) => sum + a.duration, 0);
            
            document.getElementById('stat-total-today').textContent = formatDuration(totalMins);
            document.getElementById('stat-count').textContent = dayActivities.length;

            const catTotals = {};
            let workStudyMins = 0;

            dayActivities.forEach(a => {
                catTotals[a.category] = (catTotals[a.category] || 0) + a.duration;
                if (a.category === 'Work' || a.category === 'Study') workStudyMins += a.duration;
            });
            
            // Top Cat
            let topCat = '-';
            let maxMins = 0;
            for (const [cat, mins] of Object.entries(catTotals)) {
                if (mins > maxMins) { maxMins = mins; topCat = cat; }
            }
            document.getElementById('stat-top-cat').textContent = topCat;

            // Goal Bar (Target: 6 hours / 360 mins)
            const goalMins = 360; 
            const progressPct = Math.min((workStudyMins / goalMins) * 100, 100);
            document.getElementById('goal-text').textContent = `${formatDuration(workStudyMins)} / 6h`;
            setTimeout(() => {
                document.getElementById('goal-bar').style.width = `${progressPct}%`;
            }, 100);
        }

        // --- AI Hokage Logic ---
        function generateAIInsight() {
            const titleEl = document.getElementById('ai-title');
            const msgEl = document.getElementById('ai-message');
            const iconEl = document.getElementById('ai-avatar-icon');
            
            const dayActivities = activities.filter(a => a.date === selectedDate);
            
            // Set Persona
            if (isDarkMode) {
                titleEl.textContent = "Itachi's Insight";
                iconEl.className = "fa-solid fa-eye text-red-100";
            } else {
                titleEl.textContent = "Minato's Advice";
                iconEl.className = "fa-solid fa-bolt text-yellow-100";
            }

            if (dayActivities.length === 0) {
                msgEl.textContent = isDarkMode 
                    ? '"You are wasting time. An empty day leaves you vulnerable to genjutsu."'
                    : '"Take a deep breath! Logging your first task sets the pace for a successful day."';
                return;
            }

            const totalMins = dayActivities.reduce((sum, a) => sum + a.duration, 0);
            const workMins = dayActivities.filter(a=>a.category==='Work').reduce((sum, a)=>sum+a.duration, 0);
            const studyMins = dayActivities.filter(a=>a.category==='Study').reduce((sum, a)=>sum+a.duration, 0);
            const exerciseMins = dayActivities.filter(a=>a.category==='Exercise').reduce((sum, a)=>sum+a.duration, 0);
            
            let msg = "";

            if (totalMins > 600) { // >10 hours
                msg = isDarkMode 
                    ? '"You push yourself to the brink. Is this for the village, or your own foolish ambition? Rest."'
                    : '"Incredible stamina! But remember, even the strongest Hokage needs sleep to protect the village tomorrow."';
            } else if (studyMins > workMins && studyMins > exerciseMins) {
                msg = isDarkMode
                    ? '"Knowledge without power is useless, but power without knowledge is dangerous. Keep reading the scrolls."'
                    : '"Excellent focus on studying. A sharp mind is a shinobi\'s greatest weapon!"';
            } else if (exerciseMins > 120) {
                msg = isDarkMode
                    ? '"Your Taijutsu improves. But physical strength alone won\'t stop the Akatsuki."'
                    : '"Amazing physical conditioning! Guy Sensei would be proud of your youth!"';
            } else if (workMins > 300) { // 5 hours work
                msg = isDarkMode
                    ? '"You bear the burden of the mission well. Sacrifice is the fate of a shinobi."'
                    : '"Great work ethic today! The village prospers because of hard workers like you."';
            } else {
                msg = isDarkMode
                    ? '"Balance is an illusion. Focus on your ultimate goal and eliminate distractions."'
                    : '"A balanced day! Keep up the steady pace, slow and steady wins the race."';
            }

            // Animate text change
            msgEl.style.opacity = 0;
            setTimeout(() => {
                msgEl.textContent = msg;
                msgEl.style.opacity = 1;
            }, 300);
        }

            if(!categoryChartInstance) return;
            const { textColor, gridColor } = getChartConfig();
            
            categoryChartInstance.options.plugins.legend.labels.color = textColor;
            categoryChartInstance.update();

            trendChartInstance.options.scales.x.ticks.color = textColor;
            trendChartInstance.options.scales.y.ticks.color = textColor;
            trendChartInstance.options.scales.y.grid.color = gridColor;

        function updateChartsData() {
            const dayActivities = activities.filter(a => a.date === selectedDate);

            // --- Doughnut Data ---
            const catData = {};
            dayActivities.forEach(a => catData[a.category] = (catData[a.category] || 0) + a.duration);
            categoryChartInstance.data.labels = Object.keys(catData);
            categoryChartInstance.data.datasets[0].data = Object.values(catData);
            categoryChartInstance.data.datasets[0].backgroundColor = Object.keys(catData).map(k => categoryMeta[k]?.color);
            categoryChartInstance.update();

            // --- Trend Bar Data (Last 7 Days) ---
            const trendData = {};
            const baseDate = new Date(selectedDate + "T12:00:00"); 
            for (let i = 6; i >= 0; i--) {
                const d = new Date(baseDate.getTime());
                d.setDate(d.getDate() - i);
                trendData[d.toISOString().split('T')[0]] = { label: d.toLocaleDateString('en-US', {weekday:'short'}), mins: 0 };
            }
            activities.forEach(a => { if (trendData[a.date]) trendData[a.date].mins += a.duration; });
            trendChartInstance.data.labels = Object.values(trendData).map(t => t.label);
            trendChartInstance.data.datasets[0].data = Object.values(trendData).map(t => t.mins);
            trendChartInstance.update();

            // --- Radar Chart Data (Time of Day for selected date) ---
            // Morning: 5-12, Aft: 12-17, Eve: 17-21, Night: 21-5
            let zones = [0, 0, 0, 0];
            dayActivities.forEach(a => {
                const hour = parseInt(a.startTime.split(':')[0]);
                if (hour >= 5 && hour < 12) zones[0] += a.duration;
                else if (hour >= 12 && hour < 17) zones[1] += a.duration;
                else if (hour >= 17 && hour < 21) zones[2] += a.duration;
                else zones[3] += a.duration;
            });
            radarChartInstance.data.datasets[0].data = zones;
            radarChartInstance.update();
        }
